-- users
CREATE TABLE users
(
    user_id       INT             GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    email         VARCHAR(255)    NOT NULL,
    name          VARCHAR(100)    NOT NULL,
    created_at    TIMESTAMPTZ     DEFAULT CURRENT_TIMESTAMP NOT NULL,
     PRIMARY KEY (user_id)
);
COMMENT ON TABLE users IS '사용자';
COMMENT ON COLUMN users.user_id IS '사용자 ID';
COMMENT ON COLUMN users.email IS '이메일';
COMMENT ON COLUMN users.name IS '이름';
COMMENT ON COLUMN users.created_at IS '가입일시';

-- forms
CREATE TABLE forms
(
    form_id       VARCHAR(32)     NOT NULL,
    user_id       INT             NULL,
    filename      VARCHAR(255)    NOT NULL,
    page_count    INT             DEFAULT 1 NOT NULL,
    is_form       SMALLINT        DEFAULT 1 NOT NULL,
    src_path      VARCHAR(500)    NOT NULL,
    created_at    TIMESTAMPTZ     DEFAULT CURRENT_TIMESTAMP NOT NULL,
     PRIMARY KEY (form_id)
);
COMMENT ON TABLE forms IS 'PDF 파일';
COMMENT ON COLUMN forms.form_id IS '양식 ID(UUID)';
COMMENT ON COLUMN forms.user_id IS '업로드 사용자';
COMMENT ON COLUMN forms.filename IS '원본 파일명';
COMMENT ON COLUMN forms.page_count IS '페이지 수';
COMMENT ON COLUMN forms.is_form IS 'AcroForm 여부';
COMMENT ON COLUMN forms.src_path IS '원본 저장 경로';
COMMENT ON COLUMN forms.created_at IS '업로드 일시';
ALTER TABLE forms
    ADD CONSTRAINT FK_forms_user_id_users_user_id FOREIGN KEY (user_id)
        REFERENCES users (user_id) ON DELETE RESTRICT ON UPDATE RESTRICT;

-- knowledge_documents
CREATE TABLE knowledge_documents
(
    doc_id        INT             GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    source        VARCHAR(255)    NOT NULL,
    title         VARCHAR(255)    NOT NULL,
    char_count    INT             DEFAULT 0 NOT NULL,
    indexed_at    TIMESTAMPTZ     DEFAULT CURRENT_TIMESTAMP NOT NULL,
     PRIMARY KEY (doc_id)
);
COMMENT ON TABLE knowledge_documents IS '지식 문서';
COMMENT ON COLUMN knowledge_documents.doc_id IS '문서 ID';
COMMENT ON COLUMN knowledge_documents.source IS '출처 파일명';
COMMENT ON COLUMN knowledge_documents.title IS '제목';
COMMENT ON COLUMN knowledge_documents.char_count IS '글자 수';
COMMENT ON COLUMN knowledge_documents.indexed_at IS '인덱싱 일시';

-- documents
CREATE TABLE documents
(
    document_id       INT             GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    form_id           VARCHAR(32)     NOT NULL,
    user_id           INT             NOT NULL,
    concept           TEXT            NOT NULL,
    model             VARCHAR(50)     DEFAULT 'qwen3:8b' NOT NULL,
    grounded_ratio    DECIMAL(4,3)    NOT NULL,
    status            VARCHAR(20)     DEFAULT 'generated' NOT NULL,
    output_path       VARCHAR(500)    NOT NULL,
    created_at        TIMESTAMPTZ     DEFAULT CURRENT_TIMESTAMP NOT NULL,
     PRIMARY KEY (document_id)
);
COMMENT ON TABLE documents IS '생성 문서';
COMMENT ON COLUMN documents.document_id IS '문서(작업) ID';
COMMENT ON COLUMN documents.form_id IS '대상 양식';
COMMENT ON COLUMN documents.user_id IS '요청 사용자';
COMMENT ON COLUMN documents.concept IS '사용자 입력 컨셉';
COMMENT ON COLUMN documents.model IS '사용 LLM';
COMMENT ON COLUMN documents.grounded_ratio IS '근거 일치율';
COMMENT ON COLUMN documents.status IS 'generated/filled/downloaded';
COMMENT ON COLUMN documents.output_path IS '완성 PDF 경로';
COMMENT ON COLUMN documents.created_at IS '생성 일시';
ALTER TABLE documents
    ADD CONSTRAINT FK_documents_form_id_forms_form_id FOREIGN KEY (form_id)
        REFERENCES forms (form_id) ON DELETE RESTRICT ON UPDATE RESTRICT;
ALTER TABLE documents
    ADD CONSTRAINT FK_documents_user_id_users_user_id FOREIGN KEY (user_id)
        REFERENCES users (user_id) ON DELETE RESTRICT ON UPDATE RESTRICT;

-- filled_fields
CREATE TABLE filled_fields
(
    filled_id      INT             GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    document_id    INT             NOT NULL,
    field_name     VARCHAR(100)    NOT NULL,
    value          TEXT            NOT NULL,
    grounded       SMALLINT        DEFAULT 0 NOT NULL,
    confidence     DECIMAL(4,3)    NOT NULL,
     PRIMARY KEY (filled_id)
);
COMMENT ON TABLE filled_fields IS '채운 필드';
COMMENT ON COLUMN filled_fields.filled_id IS '채운값 ID';
COMMENT ON COLUMN filled_fields.document_id IS '문서 ID';
COMMENT ON COLUMN filled_fields.field_name IS '필드명';
COMMENT ON COLUMN filled_fields.value IS '채운 값';
COMMENT ON COLUMN filled_fields.grounded IS '근거 존재 여부';
COMMENT ON COLUMN filled_fields.confidence IS '신뢰도';
CREATE UNIQUE INDEX uq_doc_field ON filled_fields(document_id, field_name);
ALTER TABLE filled_fields
    ADD CONSTRAINT FK_filled_fields_document_id_documents_document_id FOREIGN KEY (document_id)
        REFERENCES documents (document_id) ON DELETE RESTRICT ON UPDATE RESTRICT;

-- knowledge_chunks
CREATE TABLE knowledge_chunks
(
    chunk_id       VARCHAR(32)    NOT NULL,
    doc_id         INT            NOT NULL,
    chunk_index    INT            NOT NULL,
    text           TEXT           NOT NULL,
    char_count     INT            DEFAULT 0 NOT NULL,
     PRIMARY KEY (chunk_id)
);
COMMENT ON TABLE knowledge_chunks IS '지식 청크';
COMMENT ON COLUMN knowledge_chunks.chunk_id IS '청크 ID(해시)';
COMMENT ON COLUMN knowledge_chunks.doc_id IS '원본 문서 ID';
COMMENT ON COLUMN knowledge_chunks.chunk_index IS '청크 순번';
COMMENT ON COLUMN knowledge_chunks.text IS '청크 본문';
COMMENT ON COLUMN knowledge_chunks.char_count IS '글자 수';
ALTER TABLE knowledge_chunks
    ADD CONSTRAINT FK_knowledge_chunks_doc_id_knowledge_documents_doc_id FOREIGN KEY (doc_id)
        REFERENCES knowledge_documents (doc_id) ON DELETE RESTRICT ON UPDATE RESTRICT;

-- eval_results
CREATE TABLE eval_results
(
    eval_id           INT             GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    dataset           VARCHAR(50)     NOT NULL,
    accuracy          DECIMAL(4,3)    NOT NULL,
    time_reduction    DECIMAL(4,3)    NOT NULL,
    grounded_ratio    DECIMAL(4,3)    NOT NULL,
    external_calls    INT             DEFAULT 0 NOT NULL,
    measured_at       TIMESTAMPTZ     DEFAULT CURRENT_TIMESTAMP NOT NULL,
     PRIMARY KEY (eval_id)
);
COMMENT ON TABLE eval_results IS '평가 결과';
COMMENT ON COLUMN eval_results.eval_id IS '평가 ID';
COMMENT ON COLUMN eval_results.dataset IS '데이터셋명';
COMMENT ON COLUMN eval_results.accuracy IS '채움 정확도';
COMMENT ON COLUMN eval_results.time_reduction IS '시간 단축률';
COMMENT ON COLUMN eval_results.grounded_ratio IS '근거 일치율';
COMMENT ON COLUMN eval_results.external_calls IS '외부 호출 수';
COMMENT ON COLUMN eval_results.measured_at IS '측정 일시';

-- document_evidence
CREATE TABLE document_evidence
(
    filled_id    INT             NOT NULL,
    chunk_id     VARCHAR(32)     NOT NULL,
    score        DECIMAL(5,4)    NOT NULL,
     PRIMARY KEY (filled_id, chunk_id)
);
COMMENT ON TABLE document_evidence IS '근거 연결';
COMMENT ON COLUMN document_evidence.filled_id IS '채운값 ID';
COMMENT ON COLUMN document_evidence.chunk_id IS '근거 청크 ID';
COMMENT ON COLUMN document_evidence.score IS '유사도 점수';
ALTER TABLE document_evidence
    ADD CONSTRAINT FK_document_evidence_filled_id_filled_fields_filled_id FOREIGN KEY (filled_id)
        REFERENCES filled_fields (filled_id) ON DELETE RESTRICT ON UPDATE RESTRICT;
ALTER TABLE document_evidence
    ADD CONSTRAINT FK_document_evidence_chunk_id_knowledge_chunks_chunk_id FOREIGN KEY (chunk_id)
        REFERENCES knowledge_chunks (chunk_id) ON DELETE RESTRICT ON UPDATE RESTRICT;

-- form_fields
CREATE TABLE form_fields
(
    field_id      INT             GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    form_id       VARCHAR(32)     NOT NULL,
    name          VARCHAR(100)    NOT NULL,
    label         VARCHAR(255)    NOT NULL,
    field_type    VARCHAR(20)     DEFAULT 'text' NOT NULL,
    page          INT             DEFAULT 0 NOT NULL,
    bbox_x0       DECIMAL(8,2)    NOT NULL,
    bbox_y0       DECIMAL(8,2)    NOT NULL,
    bbox_x1       DECIMAL(8,2)    NOT NULL,
    bbox_y1       DECIMAL(8,2)    NOT NULL,
    required      SMALLINT        DEFAULT 0 NOT NULL,
    max_length    INT             NOT NULL,
     PRIMARY KEY (field_id)
);
COMMENT ON TABLE form_fields IS '폼 필드';
COMMENT ON COLUMN form_fields.field_id IS '필드 ID';
COMMENT ON COLUMN form_fields.form_id IS '양식 ID';
COMMENT ON COLUMN form_fields.name IS 'AcroForm 필드명';
COMMENT ON COLUMN form_fields.label IS '추론된 라벨';
COMMENT ON COLUMN form_fields.field_type IS 'text/checkbox/radio/choice';
COMMENT ON COLUMN form_fields.page IS '페이지 인덱스';
COMMENT ON COLUMN form_fields.bbox_x0 IS '좌표 x0';
COMMENT ON COLUMN form_fields.bbox_y0 IS '좌표 y0';
COMMENT ON COLUMN form_fields.bbox_x1 IS '좌표 x1';
COMMENT ON COLUMN form_fields.bbox_y1 IS '좌표 y1';
COMMENT ON COLUMN form_fields.required IS '필수 여부';
COMMENT ON COLUMN form_fields.max_length IS '최대 길이';
CREATE UNIQUE INDEX uq_form_field ON form_fields(form_id, name);
ALTER TABLE form_fields
    ADD CONSTRAINT FK_form_fields_form_id_forms_form_id FOREIGN KEY (form_id)
        REFERENCES forms (form_id) ON DELETE RESTRICT ON UPDATE RESTRICT;
